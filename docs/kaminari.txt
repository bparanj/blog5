In task model:
  paginates_per  5
  
  def self.search(term, page)
    if term
      where('name LIKE ?', "%#{term}%").order('id DESC').page(current_page)
    else
      # note: default is all, just sorted
      order('id DESC').page(current_page) 
    end
  end
